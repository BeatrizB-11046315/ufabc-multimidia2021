Código aplicado

Criação da função criar_imagens, para executar o mesmo processo em todas as imagens. Comentários sobre os procedimentos, no próprio código.

def criar_imagens(nome_arquivo):
  bgr = cv.imread(f'drive/My Drive/Colab Notebooks/{nome_arquivo}') #leitura: formato bgr
  
  # BGR para RGB
  b1, g1, r1 = cv.split(bgr)    #Os canais são separados em blue, green e red.
  rgb = cv.merge([r1,g1,b1])    #Na sequência, são rearranjados para RGB
  #Plots
  plt.figure(figsize=[12, 5])
  plt.subplot(121); plt.imshow(bgr); plt.title('BGR')
  plt.subplot(122); plt.imshow(rgb); plt.title('RBG')

  #RGB, YCrCb e HSV
  ycrcb = cv.cvtColor(bgr, cv.COLOR_BGR2YCrCb)
  hsv = cv.cvtColor(bgr, cv.COLOR_BGR2HSV)
  #Plots
  plt.figure(figsize=[15,6])
  plt.subplot(131); plt.imshow(rgb); plt.title('RGB')
  plt.subplot(132); plt.imshow(ycrcb); plt.title('YCrCb')
  plt.subplot(133); plt.imshow(hsv); plt.title('HSV')
  
  #Separação das camadas - RGB
  # Cada cor que compõe a imagem original é separada em uma imagem:
  # (imageR, imageG, imageB)
  imageR = rgb.copy()
  imageR[:, :, 1:3] = 0
  imageG = rgb.copy()
  imageG[:, :, 0] = 0; imageG[:, :, 2] = 0
  imageB = rgb.copy()
  imageB[:, :, 0:2] = 0

  #Plots
  plt.figure(figsize=[15,6])
  plt.subplot(131); plt.imshow(imageR); plt.title('RGB_Camada R')
  plt.subplot(132); plt.imshow(imageG); plt.title('RGB_Camada G')
  plt.subplot(133); plt.imshow(imageB); plt.title('RGB_Camada B')
  y1, cr1, cb1 = cv.split(ycrcb)

  #RGB, YCrCb_Y, YCrCb_Cr, YCrCb_Cb
  imageCR = ycrcb.copy()
  imageCR[:, :, 0] = 0
  imageCR[:, :, 2] = 0
  Cr = cv.cvtColor(imageCR, cv.COLOR_YCrCb2RGB)
  imageCB = ycrcb.copy()
  imageCB[:, :, 0] = 0
  imageCB[:, :, 1] = 0
  Cb = cv.cvtColor(imageCB, cv.COLOR_YCrCb2RGB)
  #Plots
  plt.figure(figsize=[15, 5])
  plt.subplot(141); plt.imshow(rgb); plt.title('RGB original')
  plt.subplot(142); plt.imshow(y1, cmap='gray'); plt.title('YCrCb_Y')
  plt.subplot(143); plt.imshow(Cr); plt.title('YCrCb_Cr')
  plt.subplot(144); plt.imshow(Cb); plt.title('YCrCb_Cb')

3. Comparar arquivos comprimidos JPEG

Código aplicado

Criação da função criar_menor_qualidade, para executar o mesmo processo em todas as imagens. Sobre IMWRITE, na documentação do openCV: "For JPEG, it can be a quality from 0 to 100 (the higher is the better). Default value is 95."

Comentários sobre os procedimentos, no próprio código.


def criar_menor_qualidade(nome_arquivo):
  bgr = cv.imread(f'drive/My Drive/Colab Notebooks/{nome_arquivo}')  # formato BGR

  #Salva com menor qualidade, fatores 25 e 5
  cv.imwrite(f'/content/drive/My Drive/Colab Notebooks/"25_{nome_arquivo}', bgr, [cv.IMWRITE_JPEG_QUALITY, 25])
  cv.imwrite(f'/content/drive/My Drive/Colab Notebooks/5_{nome_arquivo}', bgr, [cv.IMWRITE_JPEG_QUALITY, 5])

  #Leitura para visualização e conversão para acertar a cor
  rgb = cv.cvtColor(bgr, cv.COLOR_BGR2RGB)
  bgr25 = cv.imread(f'25_{nome_arquivo}'); rgb25 = cv.cvtColor(bgr25, cv.COLOR_BGR2RGB)
  bgr05 = cv.imread(f'5_{nome_arquivo}'); rgb05 = cv.cvtColor(bgr05, cv.COLOR_BGR2RGB)

  plt.figure(figsize=[15,6])
  plt.subplot(131); plt.imshow(rgb); plt.title('RGB Original')
  plt.subplot(132); plt.imshow(rgb25); plt.title('JPEG fator 25')
  plt.subplot(133); plt.imshow(rgb05); plt.title('JPEG fator 05')

#Aplicar função à imagem de todos os integrantes do grupo:

Na parte seguinte, há aplicação do código que calcula a transformada discreta de cosseno. Além disso, há a representação matricial da imagem selecionada


#Para cálculo de transformada, necessário que o valor esteja em float:
bloco8x8f = np.float32(bloco8x8)/255.0  # conversão para float
dct8x8f = cv.dct(bloco8x8f)   # calcula a DCT
dct8x8 = np.int64( (dct8x8f*255.0)) # coversão para inteiro

#Exibição dos resultados
print("(2)"); print("Imagem Y 8x8 (formato ponto flutuante)")
print( np.around(bloco8x8f, decimals = 2) )

print("(3)"); print("DCT de Y (ponto flutuante)")
print( np.around(dct8x8f, decimals = 2) )

print("(4)"); print("DCT de Y (formato inteiro)")
print(dct8x8)
